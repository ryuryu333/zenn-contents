# flake lock 生成

```
ryu@main:~$ mkdir work
ryu@main:~$ cd work
```

```
cat <<EOF > flake.nix
{
  description = "Test environment";

  inputs = {
    nixpkgs.url = "github:nixos/nixpkgs/nixpkgs-unstable";
  };

  outputs =
    { nixpkgs, ... }:
    let
      system = "x86_64-linux";
      pkgs = import nixpkgs { inherit system; };
    in
    {
      devShells.x86_64-linux.default = pkgs.mkShell {
        buildInputs = with pkgs; [
          python312
        ];
      };
    };
}
EOF
```
